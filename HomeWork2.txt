int action = 0;
Stack example = new Stack();
do
{
    Console.WriteLine("0.Вихід");
    Console.WriteLine("1.Додати елемент");
    Console.WriteLine("2.Видалити");
    Console.WriteLine("3.Показати кількість");
    Console.Write("->_");
    action = int.Parse(Console.ReadLine());
    switch(action)
    {
        case 1:
            {
                while(true)
                {
                    Console.WriteLine("Вкажіть значення(пустий рядок вихід):");
                    string item = Console.ReadLine();
                    if (string.IsNullOrEmpty(item))
                        break; //якщо пустий рядок виходимо
                    example.Push(item);
                }
                break;
            }
        case 2:
            {
                if(example.Count == 0)
                {
                    Console.WriteLine("--Стек пустий--");
                    break;
                }
                var removeItem = example.Pop();
                Console.WriteLine($"Це елемент покинув чат {removeItem}");
                break;
            }
        case 3:
            {
                Console.WriteLine($"Загальна кількість елементів {example.Count}");
                break;
            }
    }

} while (action!=0);
